```c
static const std::map<std::string,int> M_FF = {
  {"100_T",   0      },  // 0 tick!!!

  {"102_2S",  2      },  // 2
  {"103_3S",  3      },  // 3
  {"104_5S",  5      },  // 4
  {"105_10S", 10     },  // 5
  {"106_15S", 15     },  // 6
  {"107_20S", 20     },  // 7
  {"108_30S", 30     },  // 8
  
  
   //---------- for custom define 自定义周期---------------
  {"131_cu",  -1   },
  {"132_cu",  -1   },
  {"133_cu",  -1   },
  {"134_cu",  -1   },
  {"135_xp",  3456 },
  {"136_cu",  -1   },
  {"137_cu",  -1   },
  {"138_cu",  -1   },
  {"139_cu",  -1   },
  {"140_cu",  -1   },

```
